143#define CUSTOM_INSTR_R2_TYPE(funct7, funct5, rs1, funct3, opcode) \
144({ \
145 uint32_t __return; \
146 asm volatile ( \
147 "" \
148 : [output] "=r" (__return) \
149 : [input_i] "r" (rs1) \
150 ); \
151 asm volatile( \
152 ".word ( \
153 (((" #funct7 ") & 0x7f) << 25) | \

155 ((( regnum_%1 ) & 0x1f) << 15) | \
156 (((" #funct3 ") & 0x07) << 12) | \
157 ((( regnum_%0 ) & 0x1f) << 7) | \
158 (((" #opcode ") & 0x7f) << 0) \
159 );" \
