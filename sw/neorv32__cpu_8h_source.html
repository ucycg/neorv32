49inline void __attribute__ ((always_inline)) neorv32_context_save(void) {

51 // do not backup x0 and sp
52 asm volatile (
53#ifndef __riscv_32e

160
161 uint32_t reg_addr = addr;

164 asm volatile ("sw %[da], 0(%[ad])" : : [da] "r" (reg_data), [ad] "r" (reg_addr));

296/**********************************************************************/
305inline uint32_t __attribute__ ((always_inline)) neorv32_cpu_load_reservate_word(uint32_t addr) {
306
307#if defined __riscv_atomic
308 uint32_t amo_addr = addr;
309 uint32_t amo_rdata;
310
311 asm volatile ("lr.w %[dst], 0(%[addr])" : [dst] "=r" (amo_rdata) : [addr] "r" (amo_addr));
312
313 return amo_rdata;
314#else
315 (void)addr;
316
317 return 0;
318#endif
319}

322/**********************************************************************/
332inline uint32_t __attribute__ ((always_inline)) neorv32_cpu_store_conditional_word(uint32_t addr, uint32_t wdata) {
333
334#if defined __riscv_atomic
335 uint32_t amo_addr = addr;
336 uint32_t amo_wdata = wdata;
337 uint32_t amo_status;
338
339 asm volatile ("sc.w %[dst], %[src], (%[addr])" : [dst] "=r" (amo_status) : [src] "r" (amo_wdata), [addr] "r" (amo_addr));
340
341 return amo_status;
342#else
343 (void)addr;
344 (void)wdata;
345
346 return 1; // always fail
347#endif
348}
